  1. 通配符匹配

给定一个字符串 (s) 和一个字符模式 (p) ，实现一个支持 '?' 和 '*' 的通配符匹配。

动态规划

f[i,j]表示s[1-i]与p[1-j]是否匹配

分两种情况考虑，p[j]是否为‘*’，若不是，f[i,j] = p[j]==s[j]&&f[i-1, j-1]

若是，枚举'*'表示几个字符 f[i, j-1]|f[i-1, j-1]|...|f[0,j-1]，此时，时间复杂度为O(n^3)

由于 f[i-1, j] = f[i-1, j-1]|f[i-2, j-1]|...|f[0, j-1]，因此，f[i, j] = f[i, j-1] | f[i-1, j]，时间复杂度降低为O(n^2)

45.跳跃游戏

贪心无法得到最优结果，直接枚举的话时间复杂度为O(n2)。考虑动态规划f[i]表示从i到终点的最少跳数。从后往前推，此时最坏时间复杂度仍然是O(n2)。

优化

数组g[i]表示从起点到i点的最少步数。g[i]是单调递增的，分段，且每一段相差1。

可以将数组分段处理，每一段的g是相同的，下一段的边界是上一段跳跃的最远点。时间复杂度O(2n)。

class Solution {
public:
    int jump(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n);

        for(int i = 0, j = 0; i < n; i ++){
            while(j + nums[j] < i) j++;
            f[i] = f[j] + 1;
            f[0] = 0;
        }

        return f[n - 1];
    }
};

48.旋转图像

由于要求输出格式，直接按下标输出不太妥，还是要对原矩阵进行更改。如果两两个的值进行交换，比较繁琐（PS：不可行，在逐个位置进行交换的时候，有些位置会被交换两次，进而出现错误。）考虑先先沿着对角线翻转再左右翻转。

void rotate(vector<vector<int>>& matrix) {
        int m = matrix.size();
        // 先沿着对角线翻转
        for(int i = 0; i < m; i ++){
            for(int j = 0; j < i; j ++){
                swap(matrix[i][j], matrix[j][i]);
            }
        }
        // 左右翻转
        for(int i = 0; i < m;i ++){
            for(int j = 0, k = m - 1; j < k; j ++, k --){
                swap(matrix[i][j], matrix[i][k]);
            }
        }
    }

50.Pow(x, n)

快速幂。利用二进制的思想。将指数化为二进制。

x^n，若n=(1101)_2，则x^n = x(23+22+20) = x(23) * x(22) * x(20)

x(22) = x(21) * x(21)

x(23) = x(22) * x(22)。。。

class Solution {
public:
    double myPow(double x, int n) {
        typedef long long LL;
        bool is_minus = n < 0;
        double res = 1;
        // 快速幂模板  k>>=1  右移1位 
        for(LL k = abs(LL(n)); k; k >>= 1){
            if(k & 1) res *= x;
            x *= x;
        }
        if(is_minus) res = 1 / res;

        return res;
    }
};

51.N 皇后

皇后不能出现在同一行同一列以及同一对角线。

显示皇后不能在同一条对角线，需要设置一个额外的数组把点的下标映射到对角线。【难点】

class Solution {
public:
    int n;
    vector<bool> col, dg, udg;
    vector<vector<string>> ans;
    vector<string> path;

    vector<vector<string>> solveNQueens(int _n) {
        n = _n;
        col = vector<bool>(n);  //初始为false--0
        dg = udg = vector<bool>(2*n);  //注意这里，得开2n个空间，一个也不能少！
        path = vector<string>(n, string(n, '.')); //这里的初始化方法可以学习一下。

        // cout<<dg[0]<<endl;
        dfs(0);
        return ans;
    }

    void dfs(int u){
        if(u == n){
            ans.push_back(path);
            return;
        }

        for(int i = 0; i < n; i ++){
            if(!col[i] && !dg[u - i + n] && !udg[u + i]){
                col[i] = dg[u - i + n] = udg[u + i] = true;
                path[u][i] = 'Q';
                dfs(u + 1);
                path[u][i] = '.';
                col[i] = dg[u - i + n] = udg[u + i] = false;
            }
        }
    }
};

60.排列序列

全排，枚举每个位置出现哪个数字。需要深搜。

方法一：语法 next_permutation O(n * n!) 因为n比较小可以AC。

方法二：由于第一位确定的全排有(n-1)!个，分析k与全排的数目之间的规律。写法挺巧妙，注意学习。

class Solution {
public:
    string getPermutation(int n, int k) {
        string res;
        vector<bool> st(10);
        for(int i = 0; i < n; i ++){
            int fact = 1;
            for(int j = 1; j <= n - i - 1; j ++) fact *= j; // 计算阶乘

            for(int m = 1; m <= n; m ++){
                if(!st[m]){
                    if(fact < k) k -= fact;
                    else{
                        res += to_string(m); // 出现可行的数
                        st[m] = true;
                        break; // 注意每轮只能确定一个数
                    } 
                }
            }
        }
        return res;
    }
};

72.编辑距离

超级经典的DP题。

注意，这里是A转化到B，所以B是不能变化的，所以共有三种情况。只考虑最后一步（A删，A加，A修改）。

class Solution {
public:
    int minDistance(string a, string b) {
        int n = a.size(), m = b.size();
        a = " " + a, b = " " + b;
        vector<vector<int>> f(n + 1, vector<int>(m + 1, 1e8));

        for(int i = 0 ; i <= n; i ++) f[i][0] = i;
        for(int j = 1; j <= m; j ++) f[0][j] = j;

        for(int i = 1; i <= n; i ++){
            for(int j = 1; j<= m; j ++){
                f[i][j] = min(f[i - 1][j], f[i][j - 1]) + 1;
                int t = a[i] != b[j];
                f[i][j] = min(f[i][j], f[i - 1][j - 1] + t);
            }
        }

        return f[n][m];
    }
};

76.最小覆盖子串

滑动窗口（双指针）算法的经典应用。

计算以i结尾的包含所有子字符的字串的位置，起点为j，需要满足单调性。当i向后走时，j一定不会向前走。

如何快速判断某一段字符是否包含字符串中所有的字符，利用哈希表保存字符出现次数。

class Solution {
public:
    string minWindow(string s, string t) {
        unordered_map<char, int> hs, ht;
        for(auto c: t) ht[c] ++;

        string res;
        int cnt = 0;
        for(int i = 0, j = 0; i < s.size(); i ++) {
            hs[s[i]] ++;
            if(hs[s[i]] <= ht[s[i]]) cnt ++;

            while(hs[s[j]] > ht[s[j]]) hs[s[j ++]] --;
            if(cnt == t.size()){
                if(res.empty() || i - j + 1 < res.size()){
                    res = s.substr(j, i - j + 1);
                }
            }
        }

        return res;
    }
};

77.组合

组合型枚举需要注意判重。

人为规定按照某种顺序进行选择，则可以保证方案不重复。

dfs(n, u, start) // n个数，已经选了u个数，从start开始选。

class Solution {
public:

    vector<vector<int>> ans;
    vector<int> path;

    vector<vector<int>> combine(int n, int k) {
        dfs(n, k, 1);
        return ans;
    }

    void dfs(int n, int k, int start){
        if(!k){  //这里有点迷惑
            ans.push_back(path);
            return;
        }

        for(int i = start; i <= n; i ++){
            path.push_back(i);
            dfs(n, k - 1, i + 1);
            path.pop_back();
        }
    }
};

class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res;
        int n = nums.size();

        for(int i = 0; i < 1 << n; i ++){
            vector<int> path;
            for(int j = 0; j < n; j ++){ // 写法值得学习
                if(i >> j & 1) path.push_back(nums[j]);
            }
            res.push_back(path);
        }

        return res;
    }
};

class Solution {
public:
    int numTrees(int n) {
        vector<int> f(n + 1);
        f[0] = 1;

        for(int i = 1; i <= n; i ++){
            for(int j = 1; j <= i; j ++){ //遍历根节点，得到该子树可能的二叉搜索树的个数

        }

97.交错字符串

DP

f[i,j]表示s1[1-i]与s2[1-j]是否能组成s3[1-i+j]。

class Solution {
public:
    bool isInterleave(string s1, string s2, string s3) {
        int n = s1.size(), m = s2.size();
        if(s3.size() != m + n) return false;

        vector<vector<bool>> f(n + 1, vector<bool>(m + 1));

        s1 = ' ' + s1, s2 = ' ' + s2, s3 = ' ' + s3;
        for(int i = 0; i <= n; i ++){
            for(int j = 0; j <= m; j ++){
                if(!i && !j) f[i][j] = true;
                else{
                    if(i && s1[i] == s3[i + j]) f[i][j] = f[i - 1][j];
                    if(j && s2[j] == s3[i + j]) f[i][j] = f[i][j - 1] || f[i][j];
                }
            }
        } 

        return f[n][m];
    }
};

99.恢复二叉搜索树

由于二叉搜索树的中序遍历是有序的，故该题若只要求找到逆序的位置，则可以将树中序遍历后寻找逆序对。逆序对的找法则：若交换了两个相邻数的位置，则直接寻找逆序对即可，若交换了两个不相邻的数的位置，则可以寻找两个逆序对，其中，第一个逆序对的第一个数与第二个逆序对的第二个数构成逆序数对。

108.将有序数组转换为二叉搜索树

dfs

class Solution {
public:
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        return build(nums, 0, nums.size() - 1);
    }

    TreeNode* build(vector<int>& nums, int l, int r){
        if(l > r) return NULL;
        int mid = (l + r) >> 1;
        auto root = new TreeNode(nums[mid]);
        root->left = build(nums, l, mid - 1);
        root->right = build(nums, mid + 1, r);

        return root;
    }
};

110.平衡二叉树

dfs，需要记录当前子树高度的最大值，以及是否满足左子树的高度与右子树的高度最大相差1。

class Solution {
public:
    bool ans;

    bool isBalanced(TreeNode* root) {
        ans = true;
        dfs(root);
        return ans;
    }

    int dfs(TreeNode* root){ //返回当前子树高度的最大值
        if(!root) return 0;
        int lh = dfs(root->left), rh = dfs(root->right);
        if(abs(lh - rh) > 1) ans = false;
        return max(lh, rh) + 1;
    }
};

        if(!root) return 0;
        if(!root->left && !root->right) return 1;
        if(root->left && root->right) return min(minDepth(root->left), minDepth(root->right)) + 1;
        if(root->left) return minDepth(root->left) + 1;
        return minDepth(root->right) + 1;
    }
};

class Solution {
public:
    int numDistinct(string s, string t) {
        int n = s.size(), m = t.size();

        s = ' ' + s, t = ' ' + t;
        vector<vector<uint64_t>> f(n + 1, vector<uint64_t>(m + 1));//注意中间结果可能爆INT

        for(int i = 0; i <= n; i ++) f[i][0] = 1;
        for(int i = 1; i <= n; i ++) {
            for(int j = 1; j <=m; j ++){
                if(s[i] == t[j]) f[i][j] = f[i - 1][j] + f[i - 1][j -1];//当s[i]与t[j]相等时，有选s[i]和不选s[i]两种情况
                else f[i][j] = f[i - 1][j];
            }
        }

        return f[n][m];
    }
};

class Solution {
public:
    Node* connect(Node* root) {
        if(!root) return root;
        auto cur = root;
        while(cur){
            auto head = new Node(-1);
            auto tail = head;
            for(auto p = cur; p; p = p->next){
                if(p->left) tail = tail->next = p->left;
                if(p->right) tail = tail->next = p->right;
            }
            cur = head->next;
        }
        return root;
    }
};

119.杨辉三角 II

只输出某一行，若要求O(n)的空间，则可以利用滚动数组。

分析可知，若利用两个数组，则f[0]只需要存储0，2，4，6...的偶数行，f[1]存储1，3，5，7...的奇数行。

则可以利用与运算解决。

class Solution {
public:
    vector<int> getRow(int n) {
        vector<vector<int>> f(2, vector<int>(n + 1));
        for(int i = 0; i <= n; i ++){
            f[i & 1][0] = f[i & 1][i] = 1;
            for(int j = 1; j < i; j ++){
                f[i & 1][j] = f[i - 1 & 1][j - 1] + f[i - 1 & 1][j];
            }
        }
        return f[n & 1];
    }
};

class Solution {

        for(int i = f.size() - 2; i >= 0; i --){
            for(int j = 0; j <= i; j ++){
                f[i][j] += min(f[i + 1][j], f[i + 1][j + 1]);

128.最长连续序列

首先开个hash表存储所有的数，之后查x是否存在，要注意每次只枚举每一段的开头，若x - 1存在，则不枚举x。

class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int> S;
        for(auto x:nums) S.insert(x);

        int res = 0;
        for(auto x: nums){
            if(S.count(x) && !S.count(x - 1)){
                int y = x;
                S.erase(x);
                while(S.count(y + 1)){
                    y ++;
                    S.erase(y);
                }
                res = max(res, y - x + 1);
            }
        }

        return res;
    }
};

135.分发糖果

注意该题写法，由于求解f[i]的时候，需要用到f[i + 1] ，所以无法递推得到答案。

class Solution {
public:
    vector<int> f;
    vector<int> w;
    int n;

    int candy(vector<int>& ratings) {
        n = ratings.size();
        w = ratings;
        f.resize(n, -1);

        int res = 0;
        for(int i = 0; i < n; i ++){
            res += dp(i);
        }

        return res;
    }

    int dp(int x){
        if(f[x] != -1) return f[x];
        f[x] = 1;
        if(x && w[x] > w[x - 1]) f[x] = max(f[x], dp(x - 1) + 1);
        if(x + 1 < n && w[x + 1] < w[x]) f[x] = max(f[x], dp(x + 1) + 1);
        return f[x];
    }
};

138.复制带随机指针的链表

与133类似。

先复制点后复制边。

由于133的空间复杂度是O(n)。这里可以用一个取巧的方法，不需要新开哈希表。将每个点的新节点放到节点后边。

/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/

class Solution {
public:
    Node* copyRandomList(Node* head) {
        for(auto p = head; p; p = p->next->next){
            auto q = new Node(p->val);
            q->next = p->next;
            p->next = q;
        }

        // 复制random
        for(auto p = head; p; p = p->next->next){
            if(p->random){
                p->next->random = p->random->next;
            }
        }

        // 拆分两个链表
        auto dummy = new Node(-1), cur = dummy;

        for(auto p = head; p; p = p->next){
            auto q = p->next;
            cur = cur->next = q;
            p->next = q->next;
        }

        return dummy->next;
    }
};

151.翻转字符串里的单词

进阶：

  • 请尝试使用 O(1) 额外空间复杂度的原地解法。

首先将每个单词逆序，之后将整个字符串逆序。

class Solution {
public:
    string reverseWords(string s) {
        int k = 0;
        for(int i = 0; i < s.size();){
            int j = i;
            while(j < s.size() && s[j] == ' ') j ++;
            if(j == s.size()) break;   //重要！！！
            i = j;
            while(j < s.size() && s[j] != ' ') j ++;
            reverse(s.begin() + i, s.begin() + j);
            if(k) s[k ++ ] = ' ';
            while(i < j) s[k ++] = s[i ++]; // 将单词向前移动
        }
        s.erase(s.begin() + k, s.end()); //删除末尾空格
        reverse(s.begin(), s.end());

        return s;
    }
};

153.寻找旋转排序数组中的最小值

class Solution {
public:
    int findMin(vector<int>& nums) {
        int l = 0, r = nums.size() - 1;
        if(nums[r] >= nums[l]) return nums[l];
        while(l < r){
            int mid = (l + r) >> 1;
            if(nums[mid] < nums[0]) r = mid;
            else l = mid + 1;
        }
        return nums[l];
    }
};
