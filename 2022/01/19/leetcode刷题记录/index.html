  1. 通配符匹配

给定一个字符串 (s) 和一个字符模式 (p) ，实现一个支持 '?' 和 '*' 的通配符匹配。

动态规划

f[i,j]表示s[1-i]与p[1-j]是否匹配

分两种情况考虑，p[j]是否为‘*’，若不是，f[i,j] = p[j]==s[j]&&f[i-1, j-1]

若是，枚举’*’表示几个字符 f[i, j-1]|f[i-1, j-1]|…|f[0,j-1]，此时，时间复杂度为O(n^3)

由于 f[i-1, j] = f[i-1, j-1]|f[i-2, j-1]|…|f[0, j-1]，因此，f[i, j] = f[i, j-1] | f[i-1, j]，时间复杂度降低为O(n^2)

45.跳跃游戏

贪心无法得到最优结果，直接枚举的话时间复杂度为O(n^2)。考虑动态规划f[i]表示从i到终点的最少跳数。从后往前推，此时最坏时间复杂度仍然是O(n^2)。

优化

数组g[i]表示从起点到i点的最少步数。g[i]是单调递增的，分段，且每一段相差1。

可以将数组分段处理，每一段的g是相同的，下一段的边界是上一段跳跃的最远点。时间复杂度O(2n)。

class Solution {
public:
    int jump(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n);

        for(int i = 0, j = 0; i < n; i ++){
            while(j + nums[j] < i) j++;
            f[i] = f[j] + 1;
            f[0] = 0;
        }

        return f[n - 1];
    }
};

这个写法tql。简单而强大。

48. 旋转图像

由于要求输出格式，直接按下标输出不太妥，还是要对原矩阵进行更改。如果两两个的值进行交换，比较繁琐（PS：不可行，在逐个位置进行交换的时候，有些位置会被交换两次，进而出现错误。）考虑先先沿着对角线翻转再左右翻转。

void rotate(vector<vector<int>>& matrix) {
        int m = matrix.size();
        // 先沿着对角线翻转
        for(int i = 0; i < m; i ++){
            for(int j = 0; j < i; j ++){
                swap(matrix[i][j], matrix[j][i]);
            }
        }
        // 左右翻转
        for(int i = 0; i < m;i ++){
            for(int j = 0, k = m - 1; j < k; j ++, k --){
                swap(matrix[i][j], matrix[i][k]);
            }
        }
    }

50.Pow(x, n)

快速幂。利用二进制的思想。将指数化为二进制。

x^n，若n=(1101)_2，则x^n = x^(2^3+2^2+2^0) = x^(2^3) * x^(2^2) * x^(2^0)

x^(2^2) = x^(2^1) * x^(2^1)

x^(2^3) = x^(2^2) * x^(2^2)。。。

class Solution {
public:
    double myPow(double x, int n) {
        typedef long long LL;
        bool is_minus = n < 0;
        double res = 1;
        // 快速幂模板  k>>=1  右移1位 
        for(LL k = abs(LL(n)); k; k >>= 1){
            if(k & 1) res *= x;
            x *= x;
        }
        if(is_minus) res = 1 / res;

        return res;
    }
};

51.N 皇后

皇后不能出现在同一行同一列以及同一对角线。

显示皇后不能在同一条对角线，需要设置一个额外的数组把点的下标映射到对角线。【难点】

class Solution {
public:
    int n;
    vector<bool> col, dg, udg;
    vector<vector<string>> ans;
    vector<string> path;

    vector<vector<string>> solveNQueens(int _n) {
        n = _n;
        col = vector<bool>(n);  //初始为false--0
        dg = udg = vector<bool>(2*n);  //注意这里，得开2n个空间，一个也不能少！
        path = vector<string>(n, string(n, '.')); //这里的初始化方法可以学习一下。

        // cout<<dg[0]<<endl;
        dfs(0);
        return ans;
    }

    void dfs(int u){
        if(u == n){
            ans.push_back(path);
            return;
        }

        for(int i = 0; i < n; i ++){
            if(!col[i] && !dg[u - i + n] && !udg[u + i]){
                col[i] = dg[u - i + n] = udg[u + i] = true;
                path[u][i] = 'Q';
                dfs(u + 1);
                path[u][i] = '.';
                col[i] = dg[u - i + n] = udg[u + i] = false;
            }
        }
    }
};

53.最大子数组和

动态规划，f[i]表示以nums[i]结尾的子数组中的最大和。

f[i] = max(nums[i], f[i - 1] + nums[i]) = nums[i] + max(0, f[i - 1])

由于每次更新f只需要上一步的结果，所以可以直接用一个变量记录上一步的结果，不需要开新数组。

60.排列序列

全排，枚举每个位置出现哪个数字。需要深搜。

方法一：语法 next_permutation O(n * n!) 因为n比较小可以AC。

方法二：由于第一位确定的全排有(n-1)!个，分析k与全排的数目之间的规律。写法挺巧妙，注意学习。

class Solution {
public:
    string getPermutation(int n, int k) {
        string res;
        vector<bool> st(10);
        for(int i = 0; i < n; i ++){
            int fact = 1;
            for(int j = 1; j <= n - i - 1; j ++) fact *= j; // 计算阶乘

            for(int m = 1; m <= n; m ++){
                if(!st[m]){
                    if(fact < k) k -= fact;
                    else{
                        res += to_string(m); // 出现可行的数
                        st[m] = true;
                        break; // 注意每轮只能确定一个数
                    } 
                }
            }
        }
        return res;
    }
};

72.编辑距离

超级经典的DP题。

注意，这里是A转化到B，所以B是不能变化的，所以共有三种情况。只考虑最后一步（A删，A加，A修改）。

class Solution {
public:
    int minDistance(string a, string b) {
        int n = a.size(), m = b.size();
        a = " " + a, b = " " + b;
        vector<vector<int>> f(n + 1, vector<int>(m + 1, 1e8));

        for(int i = 0 ; i <= n; i ++) f[i][0] = i;
        for(int j = 1; j <= m; j ++) f[0][j] = j;

        for(int i = 1; i <= n; i ++){
            for(int j = 1; j<= m; j ++){
                f[i][j] = min(f[i - 1][j], f[i][j - 1]) + 1;
                int t = a[i] != b[j];
                f[i][j] = min(f[i][j], f[i - 1][j - 1] + t);
            }
        }

        return f[n][m];
    }
};

76.最小覆盖子串

滑动窗口（双指针）算法的经典应用。

计算以i结尾的包含所有子字符的字串的位置，起点为j，需要满足单调性。当i向后走时，j一定不会向前走。

如何快速判断某一段字符是否包含字符串中所有的字符，利用哈希表保存字符出现次数。

class Solution {
public:
    string minWindow(string s, string t) {
        unordered_map<char, int> hs, ht;
        for(auto c: t) ht[c] ++;

        string res;
        int cnt = 0;
        for(int i = 0, j = 0; i < s.size(); i ++) {
            hs[s[i]] ++;
            if(hs[s[i]] <= ht[s[i]]) cnt ++;

            while(hs[s[j]] > ht[s[j]]) hs[s[j ++]] --;
            if(cnt == t.size()){
                if(res.empty() || i - j + 1 < res.size()){
                    res = s.substr(j, i - j + 1);
                }
            }
        }

        return res;
    }
};

77.组合

组合型枚举需要注意判重。

人为规定按照某种顺序进行选择，则可以保证方案不重复。

dfs(n, u, start) // n个数，已经选了u个数，从start开始选。

写法值得学习

class Solution {
public:

    vector<vector<int>> ans;
    vector<int> path;

    vector<vector<int>> combine(int n, int k) {
        dfs(n, k, 1);
        return ans;
    }

    void dfs(int n, int k, int start){
        if(!k){  //这里有点迷惑
            ans.push_back(path);
            return;
        }

        for(int i = start; i <= n; i ++){
            path.push_back(i);
            dfs(n, k - 1, i + 1);
            path.pop_back();
        }
    }
};

class Solution {
public:
    vector<vector<int>> res;
    vector<int> path;

    vector<vector<int>> subsets(vector<int>& nums) {
        dfs(nums.size(), 1, nums);
        return res;
    }

    void dfs(int n, int start, vector<int>& nums){
        res.push_back(path);

        for(int i = start; i <= n; i ++){
            path.push_back(nums[i - 1]);
            dfs(n, i + 1, nums);
            path.pop_back();
        }
    }
};
  • 二进制
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res;
        int n = nums.size();

        for(int i = 0; i < 1 << n; i ++){
            vector<int> path;
            for(int j = 0; j < n; j ++){ // 写法值得学习
                if(i >> j & 1) path.push_back(nums[j]);
            }
            res.push_back(path);
        }

        return res;
    }
};
